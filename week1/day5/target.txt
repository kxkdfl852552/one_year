answers:
    1.STL是C++标准库里的一套通用工具，主要分三类：
        容器，负责存数据，例如vector、unordered_map、queue
        算法，负责处理数据，例如sort、find
        迭代器，负责链接容器和算法，可以简单理解成通用指针
    2.vector可以理解成一个可以自动扩容的数组
        特点：
            元素在内存中连续存放
            支持下标随机访问
            尾部插入效率高
            空间不够时自动扩容
        基本使用：
            需要头文件  #include <vector>
            创建vector
                vector<int> v;
                vector<int> v = {1,2,3,4};  //创建并初始化
                vector<int> v(10);  //创建10个元素，默认初始化为0
                vector<int> v(10, 5);  //创建10个元素，每个元素都是5
            尾部添加元素
                v.push_back(10);
            访问元素
                使用下标  cout << v[0] << endl;
                使用at() cout << v.at(0) << endl;
                v[100] 越界时属于未定义行为，可能崩溃，也可能读到垃圾数据。
                v.at(100) 越界时会抛出异常，更安全，但多了一点检查开销。
                性能敏感、下标确定合法：使用 []
                外部输入、下标不确定：使用 at()
            获取首尾元素
                cout << v.front() << endl;
                cout << v.back() << endl;
                注意：空 vector 调用 front() 或 back() 是未定义行为。应该先判断非空
            获取元素数量
                cout << v.back() << endl;  //也需要用v.empty()判断
            遍历 vector
                for (int x : v) {
                    cout << x << " ";  // 这里 x 是每个元素的副本，修改 x 不会修改原来的元素。
                }
                for (int& x : v) {
                    x = 0; //这里可以修改
                }
                for (const int& x : v) {
                    cout << x << endl; //只读不复制，推荐使用
                }
            删除尾部元素
                v.pop_back();
                删除最后一个元素，不反悔被删除的元素
                可以先调用v.back()再v.pop_back()来保留被删除的最后一个元素，同样需要判断v非空
            清空元素
                v.clear()；  //会把元素数量变成0，但通常不会释放已经申请的底层内存
            底层是什么
                底层是一个连续的动态数组
                可以粗略理解为内部保存了三个指针：start  finish  end_of_storage
                    start           指向第一个元素
                    finish          指向最后一个有效元素之后
                    end_of_storage  指向已申请内存的末尾
                    所以size()标识当前有多少有效元素  capacity()标识当前申请的内存最多能容纳多少个元素
            为什么可以随机访问
                因为内存连续存放
            vector扩容会发生什么
                push_back的时候容量不够会发生
                    申请一块更大的连续内存
                    把旧元素拷贝或移动到新内存
                    在新内存中构造新元素
                    销毁旧内存中的元素
                    释放旧内存
                    更新内部指针
                扩容后原来的地址失效，扩容前的裸指针可能成为悬空指针
            reserver和resize
                reserver 
                    vector<int> v;
                    v.reserve(100); //提前申请至少能容纳xx个元素的空间
                    此时size为0，capacity至少为100
                    v[0] = 100; //不能这样，size是0，应该 v.push_back(100);
                resize 
                    vector<int> v;
                    v.resize(100); //此时真的创建了 100 个元素
                reserve：只预留内存，不创建元素
                resize：改变有效元素数量，会创建或销毁元素
            常见坑
                循环中频繁扩容，已知数据量时应该提前reserve
                错误使用 reserve：只预留内存，不创建元素
                扩容导致引用和指针失效
                边遍历边删除
                    下面的写法有问题
                    for(int it = v.begin(); it != v.end(); ++it){
                        if(*it == 3){
                            v.erase(it);
                        }
                    }
                    erase后当前迭代器失效
                    正确写法
                    for(int it = v.begin(); it != v.end(); ++it){
                        if(*it == 3){
                            it = v.erase(it);
                        }else{
                            ++it;
                        }
                    }
                    删除所有满足条件的元素，可以使用erase-remove
                    v.erase(remove(v.begin(), v.end(), 3), v.end());
                使用 int 接收 size, 应该使用无符号类型 size_t
    3.unordered_map是一个基于哈希表实现的键值容器

    代码：

    #include <iostream>
    #include <vector>
    #include <unordered_map>
    #include <queue>
    #include <algorithm>
    using namespace std;

    int main() {
        vector<int> v = {5, 1, 3, 2, 4};
        sort(v.begin(), v.end());

        for (int x : v) {
            cout << x << " ";
        }
        cout << endl;

        unordered_map<string, int> mp;
        mp["cuda"] = 1;
        mp["kernel"] = 2;

        cout << mp["cuda"] << endl;

        priority_queue<int> pq;
        pq.push(3);
        pq.push(10);
        pq.push(1);

        cout << pq.top() << endl;

        return 0;
    }
